# plugin name
set(Output onepad)
if(NOT MSVC)
set(onepadFinalFlags
    -fvisibility=hidden
    -Wall
    -Wno-parentheses
    )
endif()

# onepad sources
set(onepadSources
    controller.cpp
    onepad.cpp
    state_management.cpp)
 # onepad headers
 set(onepadHeaders
     bitwise.h
     controller.h
     GamePad.h
     KeyStatus.h
     onepad.h
     state_management.h)

set(onepadFinalLibs
    ${onepadFinalLibs}
)

set(onepadFinalSources
    ${onepadSources}
    ${onepadHeaders}
)

include_directories(
    ${CMAKE_BINARY_DIR}/plugins/onepad/
)

if(NOT GTKn_FOUND)
   include_directories(${GLIB_INCLUDE_DIRS})
endif()


if(BUILTIN_PAD)
    add_pcsx2_lib(${Output} "${onepadFinalSources}" "${onepadFinalLibs}" "${onepadFinalFlags}")
else()
    add_pcsx2_plugin(${Output} "${onepadFinalSources}" "${onepadFinalLibs}" "${onepadFinalFlags}")
endif()

target_compile_features(${Output} PRIVATE cxx_std_17)
